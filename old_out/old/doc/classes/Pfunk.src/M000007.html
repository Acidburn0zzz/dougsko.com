        def print_template(file)
                template ="<% require 'rubygems' %>                \n<% require 'pfunk' %>\n<% funk = Pfunk.new %>\n\n<% puts funk.print_head %>\n<body>\n\n<div id=\"heading\">\n        <a href=\"index.rhtml\">\#{@title}</a>\n</div>\n\n<div id=\"navigation\">\n<%\nfunk.list_files do |full, nice|\n        puts \"<a href=\\\#{full}>\\\#{nice}</a><br />\"\nend\n%>\n</div>\n\n<% text =<<EOF\nh2. TEXT GOES HERE\n\nRuby: Optimized for programmer happiness.\n\n\\EOF\n%>\n\n<div id=\"centerDoc\">\n<% puts funk.reddify(\"\\\#{text}\") %>\n</div>\n\n<div id=\"footer\">\n        <% me = `basename \\\#{__FILE__}` %>\n        <% puts \"<a href='\\\#{me}.txt'>view source</a>\" %>\n</div>\n\n</body>\n</html>\n"

                @filename = file + ".rhtml"
                new = File.open(@filename, "w")
                new.puts template
                new.close
        end
